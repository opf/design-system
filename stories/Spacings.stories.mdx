import { Meta, Story, Canvas } from '@storybook/addon-docs';
import tokens from '../packages/styles/tokens/dist/tokens.json';

export const SpacingPreview = ({ tokens }) => <div className="spacings">
  {Object.keys(tokens)
  .filter(key => key.startsWith('s-'))
  .map(key => ({
    i: parseInt(key.split('-')[1], 10),
    name: key,
  }))
  .sort((a, b) => a.i - b.i)
  .map(({ name }) => (
    <div
      className="spacings--spacing"
      key={name}
    >
      <div className="spacings--name">{name}</div>
      <div className="spacings--value">{tokens[name]}</div>
      <div
        className="spacings--preview"
        style={{ 'height': tokens[name] }}
      />
    </div>
  ))}
</div>

<Meta title="Tokens/Spacings" />
